else if else if используется в условных операторах для проверки дополнительных условий, если предыдущее условие (в if) оказалось ложным.

Циклы
цикл что из себя представляет
for Используется, когда известно, сколько раз нужно выполнить цикл.
while Выполняется, пока условие истинно (true). Этот цикл удобен, если количество итераций заранее неизвестно.
do-while. Похож на while, но его ключевая особенность в том, что тело цикла выполняется хотя бы один раз, даже если условие ложно (точнее равно false).

Задания для закрепления

1) Сумма четных чисел: Напишите программу, которая находит сумму всех четных чисел от 1 до N, где N вводит пользователь.

#include
using namespace std;
int main() {
int N, sum = 0;
cout << "Введите число N: ";
cin >> N;
for (int i = 1; i <= N; i++) {
if (i % 2 == 0) {
sum += i;
}
}
cout << "Сумма четных чисел от 1 до " << N << " равна: " << sum << endl;
return 0;
}

2)Факториал числа.Напишите программу, которая вычисляет факториал числа N, введенного пользователем.

#include
using namespace std;
int main() {
int N;
unsigned long long factorial = 1;
cout << "Введите число N: ";
cin >> N;
for (int i = 1; i <= N; i++) {
factorial *= i;
}
cout << "Факториал числа " << N << " равен: " << factorial << endl;
return 0;
}

3)Числа Фибоначчи. Напишите программу, оторая выводит первые N чисел Фибоначчи, где N вводит пользователь.

#include
using namespace std;
int main() {
int N;
cout << "Введите количество чисел Фибоначчи: ";
cin >> N;
int a = 0, b = 1;
cout << "Числа Фибоначчи: ";
for (int i = 1; i <= N; i++) {
cout << a << " ";
int next = a + b;
a = b;
b = next;
}
cout << endl;
return 0;
}

4) Проверка на простоту. Напишите программу, которая проверяет, является ли введенное пользователем число простым.

#include
#include
using namespace std;
int main() {
int N;
bool isPrime = true;
cout << "Введите число N: ";
cin >> N;
if (N <= 1) {
isPrime = false;
} else {
for (int i = 2; i <= sqrt(N); i++) {
if (N % i == 0) {
isPrime = false;
break;
}
}
}

if (isPrime) {
cout << N << " является простым числом." << endl;
} else {
cout << N << " не является простым числом." << endl;
}

return 0;
}

5)Обратный порядок. Напишите программу, которая запрашивает у пользователя N чисел и выводит их в обратном порядке.

#include
using namespace std;
int main() {
int N;
cout << "Введите количество чисел: ";
cin >> N;
int* numbers = new int[N];
cout << "Введите " << N << " чисел:" << endl;
for (int i = 0; i < N; i++) {
cin >> numbers[i];
}
cout << "Числа в обратном порядке: ";
for (int i = N - 1; i >= 0; i--) {
cout << numbers[i] << " ";
}
cout << endl;
delete[] numbers;
return 0;
}
